package basics

import (
	"fmt"
	foo "net/http"
)

func main() {
	fmt.Println("Hello, World!")

	res, err := foo.Get("https://jsonplaceholder.typicode.com/posts/1")
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	defer res.Body.Close()

	fmt.Println("Response Status:", res.Status)
	fmt.Println("Response Body:", res.Body)
}
